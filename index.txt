// ── state ────────────────────────────────────────────────────
let view = "home"; // home | form | success | dashboard

// ── firebase listener ────────────────────────────────────────
onValue(ref(db, "checkins"), snap => {
  allEntries = [];
  if (snap.exists()) {
    snap.forEach(child => allEntries.push({ _key: child.key, ...child.val() }));
  }
  dbLoaded = true;
  render();
});

// ── helpers ──────────────────────────────────────────────────
function initials(name) {
  return name.trim().split(" ").map(w => w[0]).filter(Boolean).slice(0,2).join("").toUpperCase();
}
function pillClass(v) { return v <= 2 ? "pill pill-bad" : v === 3 ? "pill pill-mid" : "pill pill-ok"; }
function barColor(v) { return v <= 2 ? "#e74c3c" : v === 3 ? "#C0A258" : "#304E38"; }
function avg(arr, key) {
  if (!arr.length) return "—";
  return (arr.reduce((s,d) => s + d[key], 0) / arr.length).toFixed(1);
}

// ── render ───────────────────────────────────────────────────
function render() {
  document.getElementById("app").innerHTML = buildHTML();
  attachEvents();
}

function buildHTML() {
  if (view === "home")      return buildHome();
  if (view === "form")      return buildForm();
  if (view === "success")   return buildSuccess();
  if (view === "dashboard") return buildDashboard();
}

  </div>`;
}

function buildDashboard() {
  const cats = ["Todas","Sub-15","Sub-17","Sub-20"];
  const filtered = catFilter === "Todas" ? allEntries : allEntries.filter(d => d.cat === catFilter);
  const alerts = filtered.filter(d => d.q2 <= 2 || d.q3 <= 2);

  // group by date
  const grouped = {};
  filtered.forEach(d => {
    if (!grouped[d.date]) grouped[d.date] = [];
    grouped[d.date].push(d);
  });
  const dates = Object.keys(grouped).sort((a,b) => {
    const p = s => s.split("/").reverse().join("-");
    return p(b).localeCompare(p(a));
  });

  return `
  <button class="back-btn" id="btn-back">← Voltar</button>
  <div class="screen">
    <div class="dash-header">
      <div>
        <p style="font-size:11px;color:#8CA591;font-weight:500;letter-spacing:1px;text-transform:uppercase;margin:0">Painel</p>
        <p style="font-size:20px;font-weight:500;color:#304E38;margin:0">Check-in Pré-Jogo</p>
      </div>
      <div class="dash-actions">
        <button class="btn-danger" id="btn-clear">Limpar dados</button>
      </div>
    </div>

    <div class="filter-row">
      ${cats.map(c => `<button class="filter-btn ${catFilter===c?"active":""}" data-cat="${c}">${c}</button>`).join("")}
    </div>

    <div class="metric-grid">
      <div class="metric"><p class="mlabel">Respostas</p><p class="mval">${filtered.length}</p></div>
      <div class="metric"><p class="mlabel">Média preparo</p><p class="mval">${avg(filtered,"q1")}</p></div>
      <div class="metric"><p class="mlabel">Média emocional</p><p class="mval">${avg(filtered,"q2")}</p></div>
      <div class="metric ${alerts.length>0?"alert":""}"><p class="mlabel">Alertas</p><p class="mval">${alerts.length}</p></div>
    </div>

    ${alerts.length > 0 ? `
    <div class="alert-box">
      <p class="atitle">⚠ Atenção individual recomendada</p>
      ${alerts.map(a => `<p>${a.name} (${a.cat}) — emocional: ${a.q2} · foco: ${a.q3}${a.obs ? ` · <em>"${a.obs}"</em>` : ""}</p>`).join("")}
    </div>` : ""}

    ${!dbLoaded ? `<p class="loading">Carregando...</p>` :
      filtered.length === 0 ? `<div class="empty">Nenhuma resposta ainda.<br>Compartilhe o link com os atletas.</div>` :
      dates.map(date => `
        <p class="date-label">${date}</p>
        ${grouped[date].map(d => `
          <div class="athlete-card" data-key="${d._key}">
            <div class="athlete-row">
              <div style="display:flex;align-items:center;gap:10px">
                <div class="avatar"><span>${initials(d.name)}</span></div>
                <div>
                  <p class="athlete-name">${d.name}</p>
                  <p class="athlete-sub">${d.cat} · ${d.time}</p>
                </div>
              </div>
              <div class="pills">
                <span class="${pillClass(d.q1)}">${d.q1}</span>
                <span class="${pillClass(d.q2)}">${d.q2}</span>
                <span class="${pillClass(d.q3)}">${d.q3}</span>
                <span class="${pillClass(d.q4)}">${d.q4}</span>
                <span class="expand-icon">${expanded===d._key?"▲":"▼"}</span>
              </div>
            </div>
            ${expanded===d._key ? `
            <div class="detail">
              ${[["Preparo","q1"],["Estado emocional","q2"],["Foco","q3"],["Confiança","q4"]].map(([lbl,k]) => `
                <div class="bar-row">
                  <span class="bar-label">${lbl}</span>
                  <div class="bar-track"><div class="bar-fill" style="width:${d[k]/5*100}%;background:${barColor(d[k])}"></div></div>

                </div>`).join("")}
              ${d.obs ? `<div class="obs-box">💬 ${d.obs}</div>` : ""}
            </div>` : ""}
          </div>`).join("")}
      `).join("")
    }
  </div>`;
}

// ── events ───────────────────────────────────────────────────
function attachEvents() {
  // home
  document.querySelectorAll(".cat-btn[data-cat]").forEach(btn => {
    btn.addEventListener("click", () => {
      formCat = btn.dataset.cat;
      scores = { q1:0, q2:0, q3:0, q4:0 };
      nameVal = ""; obsVal = ""; formError = "";
      view = "form"; render();
    });
  });
  document.getElementById("btn-dash")?.addEventListener("click", () => { view="dashboard"; render(); });

  // form
  document.getElementById("btn-back")?.addEventListener("click", () => { view="home"; render(); });
  document.getElementById("inp-name")?.addEventListener("input", e => { nameVal = e.target.value; });
  document.getElementById("inp-obs")?.addEventListener("input", e => { obsVal = e.target.value; });
  document.querySelectorAll(".scale-btn[data-qid]").forEach(btn => {
    btn.addEventListener("click", () => {
      scores[btn.dataset.qid] = parseInt(btn.dataset.val);
      render();
    });
  });
  document.getElementById("btn-submit")?.addEventListener("click", async () => {
    nameVal = document.getElementById("inp-name")?.value || nameVal;
    obsVal  = document.getElementById("inp-obs")?.value  || obsVal;
    if (!nameVal.trim())                             { formError = "Por favor, informe seu nome."; render(); return; }
    if (Object.values(scores).some(v => v === 0))   { formError = "Responda todas as perguntas antes de enviar."; render(); return; }
    const entry = {
      cat: formCat, name: nameVal.trim(),
      date: new Date().toLocaleDateString("pt-BR"),
      time: new Date().toLocaleTimeString("pt-BR",{hour:"2-digit",minute:"2-digit"}),
      q1: scores.q1, q2: scores.q2, q3: scores.q3, q4: scores.q4,
      obs: obsVal.trim(),
    };
    await push(ref(db, "checkins"), entry);
    successName = nameVal.trim();
    view = "success"; render();
  });

  // dashboard
  document.querySelectorAll(".filter-btn[data-cat]").forEach(btn => {
    btn.addEventListener("click", () => { catFilter = btn.dataset.cat; render(); });
  });
  document.querySelectorAll(".athlete-card[data-key]").forEach(card => {
    card.addEventListener("click", () => {
      expanded = expanded === card.dataset.key ? null : card.dataset.key;
      render();
    });
  });
  document.getElementById("btn-clear")?.addEventListener("click", async () => {
    if (confirm("Apagar todas as respostas? Esta ação não pode ser desfeita.")) {
      await remove(ref(db, "checkins"));
    }
  });
}

render();
